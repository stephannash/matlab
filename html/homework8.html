Question #3

A = [-3 4 7;9 11 -13;-17 19 -23;-29 31 37;39 41 47];
[row,colm] = size(A);

%Put matrix into array
arrayA = [];
k = 1;
for i = 1:1:row

    for j = 1:1:colm

      arrayA(k) = A(i,j);
      k = k+1;

    end

end

%sort array
curMin = arrayA(1);
newA = [];
while length(arrayA) > 0
    curMin = arrayA(1);
    temp = 1;
    for j = 1:1:length(arrayA)
        if curMin > arrayA(j);
            curMin = arrayA(j);
            temp = j;
        end
    end
    newA = [newA curMin];
    arrayA(temp) = [];
end

%Put array back into matrix
k = 1;
for i = 1:1:row

    for j = 1:1:colm

      A(i,j) = newA(k);
      k = k+1;

    end

end
A

   -13    -3     4
     7     9    11
