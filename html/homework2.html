Chapter 11 #1

r = 3;
%t = [0:((2*pi)/1000)+1/1000000:2*pi];
t = linspace(0,2*pi,1000);
x = r*(t - sin(t));
y = r*(1 - cos(t));


figure(1)
hold on
plot(x,y)
xlabel('x-axis')
ylabel('y-axis')
grid on
ylim([0,7])
xlim([-1,20])
title('Cycloid')
hold off

Chapter 11 #3

x = linspace(0,5,1000);
y1 = 3 + exp(-x) .* sin(6*x);
y2 = 4 + exp(-x) .* cos(6*x);

figure(2)
hold on
plot(x,y1)
plot(x,y2)
title('Two Functions on an Single Axis')
xlabel('x-axis')
ylabel('y-axis')
legend('y1(x)','y2(x)','location','NorthWest')
hold off
